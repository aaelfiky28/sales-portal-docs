+  async getDeveloperResales(developerId: number) {
+    const sales = await this.salesRepository.findMany({
+      where: {
+        developerId: developerId,
+        saleCategory: 'Developer Resale',
+        unitValue: { not: null },
+        unitNumber: { not: null },
+        dateOfContraction: { gte: new Date('2024-01-01') },
+        SaleRevenue: {
+          basicRevenue: { gt: 0 },
+        },
+        deleted: false,
+      },
+      include: {
+        InvoiceCollections: true,
+        SaleRevenue: {
+          include: {
+            SaleRevenueSlab: true,
+            SaleRevenueIncentive: true,
+          },
+        },
+      },
+    });
+
+    return sales
+      .filter((sale) => {
+        if (sale.deleted) return false;
+        if (!sale.SaleRevenue?.claimedOrInvoiced) return true;
+        const basicRevenue = sale.SaleRevenue?.basicRevenue || 0;
+        const slabRevenue =
+          sale.SaleRevenue?.SaleRevenueSlab.reduce(
+            (acc, curr) => acc + curr.amount,
+            0,
+          ) || 0;
+        const incentiveAmount =
+          sale.SaleRevenue?.SaleRevenueIncentive.reduce(
+            (acc, curr) => acc + curr.amount,
+            0,
+          ) || 0;
+        const totalRevenue = basicRevenue + slabRevenue + incentiveAmount;
+        const totalClaimedOrInvoiced =
+          (sale.SaleRevenue?.claimedOrInvoiced || 0) +
+          (sale.SaleRevenue?.claimedOrInvoicedIncentive || 0);
+
+        const isPartialClaimedOrInvoiced =
+          totalClaimedOrInvoiced > 0 &&
+          totalClaimedOrInvoiced + 100 < totalRevenue;
+
+        const isDirectPayment = sale.InvoiceCollections?.some((c) =>
+          c.serialNumber?.includes('default_invoice'),
+        );
+
+        return isPartialClaimedOrInvoiced && isDirectPayment;
+      })
+      .map((sale) => new DirectPaymentDeveloperSaleResponseDto(sale));
+  }
