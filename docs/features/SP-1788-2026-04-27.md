import {
  PartnersRevenueSchemeStatus,
  Prisma,
} from '@prisma/client';

/**
 * INSERT/DELETE trigger shape: `{ new: row }` or `{ old: row }`.
 */
export function extractRowFromVersionValues(
  values: Prisma.JsonValue,
  key: 'new' | 'old',
): Record<string, unknown> | null {
  if (!isPlainObject(values)) return null;
  const payload = values[key];
  if (!isPlainObject(payload)) return null;
  return { ...payload };
}

/**
 * Merge UPDATE audit delta: `{ col: { old, new } }` into a snake_case row.
 */
export function mergeUpdateDeltaIntoRow(
  row: Record<string, unknown>,
  delta: Prisma.JsonValue,
): void {
  if (!isPlainObject(delta)) return;
  for (const [col, change] of Object.entries(delta)) {
    if (!SCHEME_SNAKE_KEYS.has(col)) continue;
    if (!isPlainObject(change)) continue;
    if (!('new' in change) && !('old' in change)) continue;
    if ('new' in change) {
      row[col] = (change as { new?: unknown }).new;
    }
  }
}

/**
 * ParseStatus vlaue: PartnersRevenueSchemeStatus.
 */
export function parseStatus(v: unknown): PartnersRevenueSchemeStatus {
  if (typeof v === 'string' && v in PartnersRevenueSchemeStatus) {
    return v as PartnersRevenueSchemeStatus;
  }
  return PartnersRevenueSchemeStatus.PendingApproval;
}

/**
 * Maps a snake_case row from audit JSON to Prisma unchecked create input (explicit id allowed).
 */
export function mapSnakeRowToPartnersRevenueSchemeUncheckedCreate(
  row: Record<string, unknown>,
): Prisma.PartnersRevenueSchemeUncheckedCreateInput {
  const id = parseIntVal(row.id);
  if (id === undefined) {
    throw new Error('Audit row missing id');
  }
  
  const partnersAllowedTierId = parseIntVal(row.partners_allowed_tier_id);
  const subContractId = parseIntVal(row.subcontract_id);
  const compoundId = parseIntVal(row.compound_id);
  if (
    partnersAllowedTierId === undefined ||
    subContractId === undefined ||
    compoundId === undefined
  ) {
    throw new Error(
      'Audit row missing partners_allowed_tier_id, subcontract_id, or compound_id',
    );
  }
  
  const tierId =
    typeof row.tier_id === 'string'
      ? row.tier_id
      : row.tier_id != null
        ? String(row.tier_id)
        : '';
  if (!tierId) {
    throw new Error('Audit row missing tier_id');
  }
  
  const commissionTypeAText =
    typeof row.commission_type_a_text === 'string'
      ? row.commission_type_a_text
      : String(row.commission_type_a_text ?? '');
  const commissionTypeBText =
    typeof row.commission_type_b_text === 'string'
      ? row.commission_type_b_text
      : String(row.commission_type_b_text ?? '');
  
  const createdAt = parseDate(row.created_at) ?? new Date();
  const updatedAt = parseDate(row.updated_at) ?? new Date();
  
  const data: Prisma.PartnersRevenueSchemeUncheckedCreateInput = {
    id,
    partnersAllowedTierId: Number(partnersAllowedTierId),
    tierId,
    subContractId,
    compoundId,
    commissionTypeAText,
    commissionTypeAVal: parseFloatVal(row.commission_type_a_val) ?? 0,
    commissionTypeBText,
    commissionTypeBVal: parseFloatVal(row.commission_type_b_val) ?? 0,
    commissionTypeCText:
      row.commission_type_c_text != null
        ? String(row.commission_type_c_text)
        : 'Gross',
    commissionTypeCVal: parseFloatVal(row.commission_type_c_val) ?? 0,
    currentSlab: parseFloatVal(row.current_slab),
    currentIncentive: parseFloatVal(row.current_incentive),
    currentBasicUpgrade: parseFloatVal(row.current_basic_upgrade),
    status: parseStatus(row.status),
    approvedBy:
      row.approved_by === null || row.approved_by === undefined
        ? null
        : String(row.approved_by),
    approvedOn: parseDate(row.approved_on) ?? null,
    isManual: parseBool(row.is_manual) ?? false,
    deleted: true,
    createdAt,
    updatedAt,
    ineligibilityExpressReason:
      row.ineligibility_express_reason == null
        ? null
        : String(row.ineligibility_express_reason),
  };
  
  return data;
}

/**
 * Full scheme row (after INSERT + UPDATE merges) → unchecked update payload.
 */
export function mapSnakeRowToPartnersRevenueSchemeUncheckedUpdateFromRow(
  row: Record<string, unknown>,
): { id: number; data: Prisma.PartnersRevenueSchemeUncheckedUpdateInput } {
  const full = mapSnakeRowToPartnersRevenueSchemeUncheckedCreate(row);
  const { id, createdAt: _createdAt, ...rest } = full;
  return {
    id,
    data: stripUndefined(
      rest as unknown as Record<string, unknown>,
    ) as Prisma.PartnersRevenueSchemeUncheckedUpdateInput,
  };
}
